16""" 

17This module implements support for arbitrary floating point numbers in 

18Conjecture. It doesn't make any attempt to get a good distribution, only to 

19get a format that will shrink well. 

20 

21It works by defining an encoding of non-negative floating point numbers 

22(including NaN values with a zero sign bit) that has good lexical shrinking 

23properties. 

24 

25This encoding is a tagged union of two separate encodings for floating point 

26numbers, with the tag being the first bit of 64 and the remaining 63-bits being 

27the payload. 

28 

29If the tag bit is 0, the next 7 bits are ignored, and the remaining 7 bytes are 

30interpreted as a 7 byte integer in big-endian order and then converted to a 

31float (there is some redundancy here, as 7 * 8 = 56, which is larger than the 

32largest integer that floating point numbers can represent exactly, so multiple 

33encodings may map to the same float). 

34 

35If the tag bit is 1, we instead use something that is closer to the normal 

36representation of floats (and can represent every non-negative float exactly) 

37but has a better ordering: 

38 

391. NaNs are ordered after everything else. 

402. Infinity is ordered after every finite number. 

413. The sign is ignored unless two floating point numbers are identical in 

42 absolute magnitude. In that case, the positive is ordered before the 

43 negative. 

444. Positive floating point numbers are ordered first by int(x) where 

45 encoding(x) < encoding(y) if int(x) < int(y). 

465. If int(x) == int(y) then x and y are sorted towards lower denominators of 

47 their fractional parts. 

48 

49The format of this encoding of floating point goes as follows: 

50 

51 [exponent] [mantissa] 

52 

53Each of these is the same size their equivalent in IEEE floating point, but are 

54in a different format. 

55 

56We translate exponents as follows: 

57 

58 1. The maximum exponent (2 ** 11 - 1) is left unchanged. 

59 2. We reorder the remaining exponents so that all of the positive exponents 

60 are first, in increasing order, followed by all of the negative 

61 exponents in decreasing order (where positive/negative is done by the 

62 unbiased exponent e - 1023). 

63 

64We translate the mantissa as follows: 

65 

66 1. If the unbiased exponent is <= 0 we reverse it bitwise. 

67 2. If the unbiased exponent is >= 52 we leave it alone. 

68 3. If the unbiased exponent is in the range [1, 51] then we reverse the 

69 low k bits, where k is 52 - unbiased exponent. 

70 

71The low bits correspond to the fractional part of the floating point number. 

72Reversing it bitwise means that we try to minimize the low bits, which kills 

73off the higher powers of 2 in the fraction first. 

74"""

79BIAS = 1023 

80MAX_POSITIVE_EXPONENT = MAX_EXPONENT - 1 - BIAS

85def exponent_key(e): 

86 if e == MAX_EXPONENT: 

87 return float("inf") 

88 unbiased = e - BIAS 

89 if unbiased < 0: 

90 return 10000 - unbiased 

91 else: 

92 return unbiased

95ENCODING_TABLE = array("H", sorted(range(MAX_EXPONENT + 1), key=exponent_key)) 

96DECODING_TABLE = array("H", [0]) * len(ENCODING_TABLE) 

97 

98for i, b in enumerate(ENCODING_TABLE): 

99 DECODING_TABLE[b] = i 

100 

101del i, b

104def decode_exponent(e): 

105 """Take draw_bits(11) and turn it into a suitable floating point exponent 

106 such that lexicographically simpler leads to simpler floats.""" 

107 assert 0 <= e <= MAX_EXPONENT 

108 return ENCODING_TABLE[e]

112 """Take a floating point exponent and turn it back into the equivalent

118def reverse_byte(b): 

119 result = 0 

120 for _ in range(8): 

121 result <<= 1 

122 result |= b & 1 

123 b >>= 1 

124 return result

127# Table mapping individual bytes to the equivalent byte with the bits of the 

128# byte reversed. e.g. 1=0b1 is mapped to 0xb10000000=0x80=128. We use this 

129# precalculated table to simplify calculating the bitwise reversal of a longer 

130# integer. 

131REVERSE_BITS_TABLE = bytearray(map(reverse_byte, range(256)))

134def reverse64(v): 

135 """Reverse a 64-bit integer bitwise. 

136 

137 We do this by breaking it up into 8 bytes. The 64-bit integer is then the 

138 concatenation of each of these bytes. We reverse it by reversing each byte 

139 on its own using the REVERSE_BITS_TABLE above, and then concatenating the 

140 reversed bytes. 

141 

142 In this case concatenating consists of shifting them into the right 

143 position for the word and then oring the bits together. 

144 """ 

145 assert v.bit_length() <= 64 

146 return ( 

147 (REVERSE_BITS_TABLE[(v >> 0) & 0xFF] << 56) 

148 | (REVERSE_BITS_TABLE[(v >> 8) & 0xFF] << 48) 

149 | (REVERSE_BITS_TABLE[(v >> 16) & 0xFF] << 40) 

150 | (REVERSE_BITS_TABLE[(v >> 24) & 0xFF] << 32) 

151 | (REVERSE_BITS_TABLE[(v >> 32) & 0xFF] << 24) 

152 | (REVERSE_BITS_TABLE[(v >> 40) & 0xFF] << 16) 

153 | (REVERSE_BITS_TABLE[(v >> 48) & 0xFF] << 8) 

154 | (REVERSE_BITS_TABLE[(v >> 56) & 0xFF] << 0) 

155 )

158MANTISSA_MASK = (1 << 52) - 1

161def reverse_bits(x, n): 

162 assert x.bit_length() <= n <= 64 

163 x = reverse64(x) 

164 x >>= 64 - n 

165 return x

168def update_mantissa(unbiased_exponent, mantissa): 

169 if unbiased_exponent <= 0: 

170 mantissa = reverse_bits(mantissa, 52) 

171 elif unbiased_exponent <= 51: 

172 n_fractional_bits = 52 - unbiased_exponent 

173 fractional_part = mantissa & ((1 << n_fractional_bits) - 1) 

174 mantissa ^= fractional_part 

175 mantissa |= reverse_bits(fractional_part, n_fractional_bits) 

176 return mantissa

179def lex_to_float(i): 

180 assert i.bit_length() <= 64 

181 has_fractional_part = i >> 63 

182 if has_fractional_part: 

183 exponent = (i >> 52) & ((1 << 11) - 1) 

184 exponent = decode_exponent(exponent) 

185 mantissa = i & MANTISSA_MASK 

186 mantissa = update_mantissa(exponent - BIAS, mantissa) 

187 

188 assert mantissa.bit_length() <= 52 

189 

190 return int_to_float((exponent << 52) | mantissa) 

191 else: 

192 integral_part = i & ((1 << 56) - 1) 

193 return float(integral_part)

196def float_to_lex(f): 

197 if is_simple(f): 

198 assert f >= 0 

199 return int(f) 

200 return base_float_to_lex(f)

203def base_float_to_lex(f): 

204 i = float_to_int(f) 

205 i &= (1 << 63) - 1 

206 exponent = i >> 52 

207 mantissa = i & MANTISSA_MASK 

208 mantissa = update_mantissa(exponent - BIAS, mantissa) 

209 exponent = encode_exponent(exponent) 

210 

211 assert mantissa.bit_length() <= 52 

212 return (1 << 63) | (exponent << 52) | mantissa

215def is_simple(f): 

216 try: 

217 i = int(f) 

218 except (ValueError, OverflowError): 

219 return False 

220 if i != f: 

221 return False 

222 return i.bit_length() <= 56

225def draw_float(data): 

226 try: 

227 data.start_example(DRAW_FLOAT_LABEL) 

228 f = lex_to_float(data.draw_bits(64)) 

229 if data.draw_bits(1): 

230 f = -f 

231 return f 

232 finally: 

233 data.stop_example()

236def write_float(data, f): 

237 data.draw_bits(64, forced=float_to_lex(abs(f))) 

238 sign = float_to_int(f) >> 63 

239 data.draw_bits(1, forced=sign)
