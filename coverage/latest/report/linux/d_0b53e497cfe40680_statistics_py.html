26def describe_targets(best_targets): 

27 """Return a list of lines describing the results of `target`, if any.""" 

28 # These lines are included in the general statistics description below, 

29 # but also printed immediately below failing examples to alleviate the 

30 # "threshold problem" where shrinking can make severe bug look trivial. 

31 # See https://github.com/HypothesisWorks/hypothesis/issues/2180 

32 if not best_targets: 

33 return [] 

34 elif len(best_targets) == 1: 

35 label, score = next(iter(best_targets.items())) 

36 return [f"Highest target score: {score:g} (label={label!r})"] 

37 else: 

38 lines = ["Highest target scores:"] 

39 for label, score in sorted(best_targets.items(), key=lambda x: x[::-1]): 

40 lines.append(f"{score:>16g} (label={label!r})") 

41 return lines

44def describe_statistics(stats_dict): 

45 """Return a multi-line string describing the passed run statistics. 

46 

47 `stats_dict` must be a dictionary of data in the format collected by 

48 `hypothesis.internal.conjecture.engine.ConjectureRunner.statistics`. 

49 

50 We DO NOT promise that this format will be stable or supported over 

51 time, but do aim to make it reasonably useful for downstream users. 

52 It's also meant to support benchmarking for research purposes. 

53 

54 This function is responsible for the report which is printed in the 

55 terminal for our pytest --hypothesis-show-statistics option. 

56 """ 

57 lines = [stats_dict["nodeid"] + ":\n"] if "nodeid" in stats_dict else [] 

58 prev_failures = 0 

59 for phase in ["reuse", "generate", "shrink"]: 

60 d = stats_dict.get(phase + "-phase", {}) 

61 # Basic information we report for every phase 

62 cases = d.get("test-cases", []) 

63 if not cases: 

64 continue 

65 statuses = Counter(t["status"] for t in cases) 

66 runtimes = sorted(t["runtime"] for t in cases) 

67 n = max(0, len(runtimes) - 1) 

68 lower = int(runtimes[int(math.floor(n * 0.05))] * 1000) 

69 upper = int(runtimes[int(math.ceil(n * 0.95))] * 1000) 

70 if upper == 0: 

71 ms = "< 1ms" 

72 elif lower == upper: 

73 ms = f"~ {lower}ms" 

74 else: 

75 ms = f"{lower}-{upper} ms" 

76 drawtime_percent = 100 * statistics.mean( 

77 t["drawtime"] / t["runtime"] if t["runtime"] > 0 else 0 for t in cases 

78 ) 

79 lines.append( 

80 f" - during {phase} phase ({d['duration-seconds']:.2f} seconds):\n" 

81 f" - Typical runtimes: {ms}, ~ {drawtime_percent:.0f}% in data generation\n" 

82 f" - {statuses['valid']} passing examples, {statuses['interesting']} " 

83 f"failing examples, {statuses['invalid'] + statuses['overrun']} invalid examples" 

84 ) 

85 # If we've found new distinct failures in this phase, report them 

86 distinct_failures = d["distinct-failures"] - prev_failures 

87 if distinct_failures: 

88 plural = distinct_failures > 1 

89 lines.append( 

90 " - Found {}{} distinct error{} in this phase".format( 

91 distinct_failures, " more" * bool(prev_failures), "s" * plural 

92 ) 

93 ) 

94 prev_failures = d["distinct-failures"] 

95 # Report events during the generate phase, if there were any 

96 if phase == "generate": 

97 events = Counter(sum((t["events"] for t in cases), [])) 

98 if events: 

99 lines.append(" - Events:") 

100 lines += [ 

101 f" * {100 * v / len(cases):.2f}%, {k}" 

102 for k, v in sorted(events.items(), key=lambda x: (-x[1], x[0])) 

103 ] 

104 # Some additional details on the shrinking phase 

105 if phase == "shrink": 

106 lines.append( 

107 " - Tried {} shrinks of which {} were successful".format( 

108 len(cases), d["shrinks-successful"] 

109 ) 

110 ) 

111 lines.append("") 

112 

113 target_lines = describe_targets(stats_dict.get("targets", {})) 

114 if target_lines: 

115 lines.append(" - " + target_lines[0]) 

116 lines.extend(" " + l for l in target_lines[1:]) 

117 lines.append(" - Stopped because " + stats_dict["stopped-because"]) 

118 return "\n".join(lines)
